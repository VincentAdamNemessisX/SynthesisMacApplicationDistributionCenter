info 类型, 更改状态

$('#btn-12').click(function(){
    var infoDialog = $.dialog({
        type : 'info',
        infoText : '加载中…',
        infoIcon : '{% static 'front/img/icon/loading.gif' %}'			
    });

    window.setTimeout(function() {
        infoDialog.dialog.update({
	        autoClose : 1500,
	        infoText : '操作成功',
	        infoIcon : '{% static 'front/img/icon/success.png' %}'
        });
    }, 2500);
});
